```c++
class Solution {
public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        vector<vector<int>> ans(0,vector<int>(3));
        set<vector<int>> ans_s;
        int n = nums.size();
        for(int i=0;i<n;i++){
            int target = -nums[i];
            unordered_map<int, int> v;
            for(int j=i+1;j<n;j++){
                if(v.count(target-nums[j])==0){
                    v[nums[j]]++;
                }else{
                    vector<int> temp;
                    temp.push_back(nums[i]);
                    temp.push_back(nums[j]);
                    temp.push_back(target - nums[j]);
                    sort(temp.begin(), temp.end());

                    if(ans_s.count(temp)==0){
                        ans_s.insert(temp);
                        ans.push_back(temp);
                    }
                }
            }
        }
        return ans;
    }
};
```

> 
> `set<vector<int>>` 去重很慢。  
> 有没有一种方法可以 **让重复的三元组根本不会被生成**，而不是生成之后再去重？

```c++
class Solution {
public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        vector<vector<int>> ans(0,vector<int>(3));
        set<vector<int>> ans_s;
        sort(nums.begin(), nums.end()); // 提前排序(非必要)

                    temp.push_back(nums[j]);
                    temp.push_back(target - nums[j]);

                        ans_s.insert(temp);
                        ans.push_back(temp);
                    }
                }
            }
        }
```

# 🔍 最终整理：你必须修改的地方（按重要度排序）

| 重要度   | 修改点                                                    |
| ----- | ------------------------------------------------------ |
| ⭐⭐⭐⭐⭐ | 跳过重复的 i：`if (i > 0 && nums[i] == nums[i-1]) continue;` |
| ⭐⭐⭐⭐  | temp 三元组必须排序                                           |
| ⭐⭐⭐⭐  | i > 0 直接 break：`if (nums[i] > 0) break;`               |
| ⭐⭐⭐   | 修正哈希逻辑顺序                                               |
| ⭐⭐⭐   | j 层也需要跳过重复值（你的代码缺这点，会导致重复计算）                           |
| ⭐⭐    | 若继续用哈希，不使用 set 去重（改用指针法更优）                             |
| ⭐     | 去掉无意义的 v[nums[j]]++（你只需要标记出现过，不需要计数）                   |

# 🧠 思考方式总结（帮助你下次写 3Sum）

1. **排序 + 去重（跳过重复 i / j）** 是性能关键
2. **3Sum = 固定一个值 + 求 Two Sum**
3. Two Sum 用哈希会慢，因为重复值太多，哈希不适合大量重复
4. Two Sum 更适合“排序后的双指针”
5. 三元组必须排序后存储才能去重
6. 任何可能减少循环次数的剪枝都值得加（比如 nums[i] > 0 break）

```python
class Solution:
    def threeSum(self, nums: list[int]) -> list[list[int]]:
        ans = []
        v = set()
        d = dict()
        for i in range(len(nums)):
            for j in d.keys():
                if -nums[i]-j in d.keys():
                    d[j]-=1
                    d[-nums[i]-j]-=1
                    tmp = [nums[i], j, -nums[i]-j]
                    tmp.sort()
                    s = ''.join(map(str,tmp))
                    if d[j] >= 0 and d[-nums[i]-j] >= 0 and s not in v:
                        ans.append(tmp)
                        v.add(s)
                    d[j]+=1
                    d[-nums[i]-j]+=1
            
            d[nums[i]] = d.get(nums[i], 0) + 1
        return ans
```

>正规GPT双指针版本:
>	数组 + 两数之和 +有序 → 双指针

>报错版:

	1️⃣ **找到答案后没有移动指针 → 死循环**
	
	2️⃣ **l 去重条件错误** `l - 1 > 0`
	
	3️⃣ **r 去重基本不会触发**
	
	4️⃣ **没有对 i 去重**
```python
class Solution:
    def threeSum(self, nums: list[int]) -> list[list[int]]:
        ans = []
        n = len(nums)
        nums.sort()
        for i in range(n):
            # ❌问题1：没有对 i 去重
            # 如果 nums[i] == nums[i-1]，会产生重复答案
            # threeSum 必须跳过重复的 i

            target = -nums[i]
            l, r = i + 1, n - 1
            while (l < r):
                s = nums[l] + nums[r]
                if s > target:
                    r -= 1
                elif s < target:
                    l += 1
                else:
                    ans.append([nums[i], nums[l], nums[r]])
                    # ❌问题2：找到答案后没有移动 l 和 r
                    # s 仍然等于 target
                    # 下一次循环还是同一组 nums[l] nums[r]
                    # 会不断 append，可能死循环

                # ❌问题3：l 的去重条件错误
                # l - 1 > 0 会导致很多情况下无法去重
                # 例如 l = 1 时:
                # l-1 = 0 -> 0 > 0 为 False
                # 于是不会去重
                while l - 1 > 0 and nums[l] == nums[l - 1]:
                    l += 1
                    # ❌问题4：这里没有保证 l < r
                    # 如果一直重复，l 可能越过 r

                # ❌问题5：r 的去重逻辑基本不会触发
                # 因为 r 初始是 n-1
                # r+1 = n
                # r+1 < n 一开始就是 False
                while r + 1 < n and nums[r] == nums[r + 1]:
                    r -= 1
                    # 同样没有保证 r > l

        return ans

```

```python
                    l += 1

                        r -= 1
                    r -= 1
        return ans
```

```python
class Solution:
    def threeSum(self, nums: list[int]) -> list[list[int]]:
        nums.sort()  # ✅ 先排序
        n = len(nums)
        ans = []

        for i in range(n):
            if i > 0 and nums[i] == nums[i-1]:
                continue  # 跳过重复的第一个数

            left, right = i + 1, n - 1
            while left < right:
                s = nums[i] + nums[left] + nums[right]

                if s < 0:
                    left += 1  # 和太小 → 移动左指针
                elif s > 0:
                    right -= 1  # 和太大 → 移动右指针
                else:
                    # 找到一组
                    ans.append([nums[i], nums[left], nums[right]])

                    # 去重 left 和 right
                    while left < right and nums[left] == nums[left + 1]:
                        left += 1
                    while left < right and nums[right] == nums[right - 1]:
                        right -= 1

                    # 移动指针继续搜索
                    left += 1
                    right -= 1

        return ans
```
